Crea un prototipo (es. const BoardProto = { ... }) con spostaPezzo(da, a) che:
Legge il pezzo da this[da]; se non c’è, ritorna false o lancia un Error.
Aggiorna lo stato: delete this[da]; this[a] = pezzo.
Aggiorna il DOM:
from = document.getElementById(da), to = document.getElementById(a)
Se to ha un <img>, rimuovilo (cattura).
Se from ha un <img>, sposta quel nodo dentro to (altrimenti ricrea l’immagine con PIECE_URLS[pezzo]).
Ritorna true se la mossa è andata a buon fine.
Collega il prototipo a posizioni con Object.setPrototypeOf(posizioni, BoardProto).
Prova (facoltativo): posizioni.spostaPezzo('e2','e4').

const BoardProto = {
  spostaPezzo(da, a) {
    if (!this[da]) {return false;}

    const pezzo = this[da];
    delete this[da];
    this[a] = pezzo;

    let from = document.getElementById(da);
    let to = document.getElementById(a);

    let img_to = to.querySelector("img"); //evetuale imm. d'arrivo
    let img_from = from.querySelector("img");

    if (img_to) img_to.remove(); //se esiste imm d'arrivo allora tolgo la elimino
    if (img_from) { //se c'era imm. di partenza
      to.appendChild(img_from); //metto l'imm. di partenza nella sua destinazione d'arrivo che nell' if prima ho svuotato 
    } 
    else { //se non c'era alcuna imm di partenza
      let nuovaImm = document.createElement("img"); //ne creo una nuova
      nuovaImm.src = PIECE_URLS[pezzo];
      to.appendChild(nuovaImm); //e la metto nella destinazione
    }
    return true;
  }
};

Object.setPrototypeOf(posizioni, BoardProto);
